# 03 — V4L2 摄像头采集模块

## 3.1 设计思路

将 V4L2 采集封装为独立的 C++ 类，运行在独立线程中，通过 Qt 信号槽将帧数据传递给 GUI 线程。

借鉴两个现有项目的优点：
- **从 cap-v4l2 借鉴**：VIDIOC_QUERYCAP 能力查询、VIDIOC_TRY_FMT 格式试探、select() 等待、曝光/翻转控制
- **从 VideoMonitor 借鉴**：QThread 封装、信号槽传递 QImage、JPEG/YUYV 转 RGB

### 与原 VideoMonitor 的关键区别

| 原 VideoMonitor | 新方案 |
|----------------|--------|
| 请求 V4L2_PIX_FMT_JPEG | 请求 V4L2_PIX_FMT_YUYV（STM32MP157 DCMI 通常不支持 JPEG 硬编码） |
| 依赖 libjpeg 解码 | 直接 YUYV→BGR，交给 OpenCV 处理 |
| 阻塞式 ioctl DQBUF | select() + DQBUF，带超时保护 |
| 无能力查询 | 启动时查询并打印摄像头能力、支持格式 |
| 无 V4L2 control | 支持曝光、翻转、亮度等 V4L2 control |
| 硬编码 640x480 | 分辨率可配置，启动时 TRY_FMT 验证 |

## 3.2 V4L2 采集流程

```
打开设备 /dev/video0
    ↓
VIDIOC_QUERYCAP — 查询设备能力
    ↓
VIDIOC_ENUM_FMT — 枚举支持的像素格式
    ↓
VIDIOC_TRY_FMT — 试探目标分辨率和格式
    ↓
VIDIOC_S_FMT — 设置采集格式（YUYV, 640x480）
    ↓
VIDIOC_REQBUFS — 申请内核缓冲区（4个）
    ↓
VIDIOC_QUERYBUF + mmap — 映射缓冲区到用户空间
    ↓
VIDIOC_QBUF — 将缓冲区入队
    ↓
VIDIOC_STREAMON — 开始采集
    ↓
循环：select() → VIDIOC_DQBUF → 处理帧 → VIDIOC_QBUF
    ↓
VIDIOC_STREAMOFF — 停止采集
    ↓
munmap + close — 释放资源
```

## 3.3 核心代码 — v4l2_camera.h

```cpp
#ifndef V4L2_CAMERA_H
#define V4L2_CAMERA_H

#include <QThread>
#include <QImage>
#include <string>
#include <vector>
#include <atomic>
#include <linux/videodev2.h>

struct FrameBuffer {
    void *start;
    size_t length;
};

class V4l2Camera : public QThread {
    Q_OBJECT
public:
    // 参数可配置
    V4l2Camera(const std::string &device = "/dev/video0",
               int width = 640, int height = 480,
               int bufferCount = 4);
    ~V4l2Camera();

    bool open();
    void close();
    void stop();

    // V4L2 Controls（内窥镜关键功能）
    bool setExposure(int value);
    bool setBrightness(int value);
    bool setContrast(int value);
    bool setHFlip(bool enable);
    bool setVFlip(bool enable);

    int getWidth() const { return m_width; }
    int getHeight() const { return m_height; }

signals:
    void frameReady(const QImage &image);
    void errorOccurred(const QString &msg);

protected:
    void run() override;

private:
    bool queryCapability();
    bool enumFormats();
    bool setFormat();
    bool requestBuffers();
    bool startStreaming();
    bool stopStreaming();
    bool grabFrame(void **data, int *size);
    void releaseBuffers();

    // YUYV → RGB888 转换
    void yuyvToRgb888(const unsigned char *yuyv,
                      unsigned char *rgb, int w, int h);

    // V4L2 control 辅助
    bool setControl(uint32_t id, int value);
    int  getControl(uint32_t id);

    std::string m_device;
    int m_fd;
    int m_width;
    int m_height;
    int m_bufferCount;
    std::vector<FrameBuffer> m_buffers;
    std::atomic<bool> m_running;
};

#endif // V4L2_CAMERA_H
```

## 3.4 核心代码 — v4l2_camera.cpp

```cpp
#include "v4l2_camera.h"
#include <fcntl.h>
#include <unistd.h>
#include <sys/ioctl.h>
#include <sys/mman.h>
#include <sys/select.h>
#include <cstring>
#include <QDebug>

// ioctl 重试封装
static int xioctl(int fd, int request, void *arg) {
    int r;
    do { r = ioctl(fd, request, arg); }
    while (r == -1 && errno == EINTR);
    return r;
}

V4l2Camera::V4l2Camera(const std::string &device, int width, int height, int bufferCount)
    : m_device(device), m_fd(-1), m_width(width), m_height(height),
      m_bufferCount(bufferCount), m_running(false) {}

V4l2Camera::~V4l2Camera() {
    stop();
    wait();
    close();
}

bool V4l2Camera::open() {
    m_fd = ::open(m_device.c_str(), O_RDWR | O_NONBLOCK);
    if (m_fd < 0) {
        emit errorOccurred(QString("无法打开设备: %1").arg(m_device.c_str()));
        return false;
    }
    if (!queryCapability()) return false;
    if (!enumFormats()) return false;
    if (!setFormat()) return false;
    if (!requestBuffers()) return false;
    return true;
}

void V4l2Camera::close() {
    if (m_fd >= 0) {
        stopStreaming();
        releaseBuffers();
        ::close(m_fd);
        m_fd = -1;
    }
}

void V4l2Camera::stop() {
    m_running = false;
}

bool V4l2Camera::queryCapability() {
    struct v4l2_capability cap;
    memset(&cap, 0, sizeof(cap));
    if (xioctl(m_fd, VIDIOC_QUERYCAP, &cap) < 0) {
        emit errorOccurred("VIDIOC_QUERYCAP 失败");
        return false;
    }
    qDebug() << "Driver:" << (char*)cap.driver;
    qDebug() << "Card:" << (char*)cap.card;
    qDebug() << "Bus:" << (char*)cap.bus_info;

    if (!(cap.capabilities & V4L2_CAP_VIDEO_CAPTURE)) {
        emit errorOccurred("设备不支持视频采集");
        return false;
    }
    if (!(cap.capabilities & V4L2_CAP_STREAMING)) {
        emit errorOccurred("设备不支持 streaming I/O");
        return false;
    }
    return true;
}

bool V4l2Camera::enumFormats() {
    struct v4l2_fmtdesc fmt;
    memset(&fmt, 0, sizeof(fmt));
    fmt.type = V4L2_BUF_TYPE_VIDEO_CAPTURE;
    fmt.index = 0;

    qDebug() << "支持的像素格式:";
    while (xioctl(m_fd, VIDIOC_ENUM_FMT, &fmt) == 0) {
        qDebug() << QString("  [%1] %2").arg(fmt.index).arg((char*)fmt.description);
        fmt.index++;
    }
    return true;
}

bool V4l2Camera::setFormat() {
    struct v4l2_format fmt;
    memset(&fmt, 0, sizeof(fmt));
    fmt.type = V4L2_BUF_TYPE_VIDEO_CAPTURE;
    fmt.fmt.pix.width = m_width;
    fmt.fmt.pix.height = m_height;
    fmt.fmt.pix.pixelformat = V4L2_PIX_FMT_YUYV;
    fmt.fmt.pix.field = V4L2_FIELD_NONE;

    // 先 TRY，看硬件是否支持
    if (xioctl(m_fd, VIDIOC_TRY_FMT, &fmt) < 0) {
        qDebug() << "VIDIOC_TRY_FMT 失败，尝试直接设置";
    }

    // 硬件可能调整了分辨率
    if ((int)fmt.fmt.pix.width != m_width || (int)fmt.fmt.pix.height != m_height) {
        qDebug() << QString("分辨率被调整为: %1x%2")
                        .arg(fmt.fmt.pix.width).arg(fmt.fmt.pix.height);
        m_width = fmt.fmt.pix.width;
        m_height = fmt.fmt.pix.height;
    }

    if (xioctl(m_fd, VIDIOC_S_FMT, &fmt) < 0) {
        emit errorOccurred("VIDIOC_S_FMT 失败");
        return false;
    }

    qDebug() << QString("采集格式: %1x%2 YUYV").arg(m_width).arg(m_height);
    return true;
}

bool V4l2Camera::requestBuffers() {
    struct v4l2_requestbuffers req;
    memset(&req, 0, sizeof(req));
    req.count = m_bufferCount;
    req.type = V4L2_BUF_TYPE_VIDEO_CAPTURE;
    req.memory = V4L2_MEMORY_MMAP;

    if (xioctl(m_fd, VIDIOC_REQBUFS, &req) < 0) {
        emit errorOccurred("VIDIOC_REQBUFS 失败");
        return false;
    }

    m_buffers.resize(req.count);
    for (int i = 0; i < (int)req.count; i++) {
        struct v4l2_buffer buf;
        memset(&buf, 0, sizeof(buf));
        buf.type = V4L2_BUF_TYPE_VIDEO_CAPTURE;
        buf.memory = V4L2_MEMORY_MMAP;
        buf.index = i;

        if (xioctl(m_fd, VIDIOC_QUERYBUF, &buf) < 0) {
            emit errorOccurred("VIDIOC_QUERYBUF 失败");
            return false;
        }

        m_buffers[i].length = buf.length;
        m_buffers[i].start = mmap(NULL, buf.length,
                                   PROT_READ | PROT_WRITE,
                                   MAP_SHARED, m_fd, buf.m.offset);
        if (m_buffers[i].start == MAP_FAILED) {
            emit errorOccurred("mmap 失败");
            return false;
        }

        // 入队
        if (xioctl(m_fd, VIDIOC_QBUF, &buf) < 0) {
            emit errorOccurred("VIDIOC_QBUF 失败");
            return false;
        }
    }
    return true;
}

bool V4l2Camera::startStreaming() {
    enum v4l2_buf_type type = V4L2_BUF_TYPE_VIDEO_CAPTURE;
    if (xioctl(m_fd, VIDIOC_STREAMON, &type) < 0) {
        emit errorOccurred("VIDIOC_STREAMON 失败");
        return false;
    }
    return true;
}

bool V4l2Camera::stopStreaming() {
    enum v4l2_buf_type type = V4L2_BUF_TYPE_VIDEO_CAPTURE;
    xioctl(m_fd, VIDIOC_STREAMOFF, &type);
    return true;
}

bool V4l2Camera::grabFrame(void **data, int *size) {
    fd_set fds;
    FD_ZERO(&fds);
    FD_SET(m_fd, &fds);

    struct timeval tv;
    tv.tv_sec = 2;
    tv.tv_usec = 0;

    int r = select(m_fd + 1, &fds, NULL, NULL, &tv);
    if (r <= 0) return false;

    struct v4l2_buffer buf;
    memset(&buf, 0, sizeof(buf));
    buf.type = V4L2_BUF_TYPE_VIDEO_CAPTURE;
    buf.memory = V4L2_MEMORY_MMAP;

    if (xioctl(m_fd, VIDIOC_DQBUF, &buf) < 0) return false;

    *data = m_buffers[buf.index].start;
    *size = buf.bytesused;

    // 重新入队
    if (xioctl(m_fd, VIDIOC_QBUF, &buf) < 0) return false;

    return true;
}

void V4l2Camera::releaseBuffers() {
    for (auto &b : m_buffers) {
        if (b.start && b.start != MAP_FAILED)
            munmap(b.start, b.length);
    }
    m_buffers.clear();
}

void V4l2Camera::yuyvToRgb888(const unsigned char *yuyv,
                               unsigned char *rgb, int w, int h) {
    // 每4字节 YUYV 产生2个 RGB 像素
    for (int i = 0; i < w * h / 2; i++) {
        int y0 = yuyv[0];
        int u  = yuyv[1];
        int y1 = yuyv[2];
        int v  = yuyv[3];
        yuyv += 4;

        auto clamp = [](int val) -> unsigned char {
            return val < 0 ? 0 : (val > 255 ? 255 : val);
        };

        int c0 = y0 - 16, c1 = y1 - 16;
        int d = u - 128, e = v - 128;

        rgb[0] = clamp((298 * c0 + 409 * e + 128) >> 8);           // R
        rgb[1] = clamp((298 * c0 - 100 * d - 208 * e + 128) >> 8); // G
        rgb[2] = clamp((298 * c0 + 516 * d + 128) >> 8);           // B
        rgb[3] = clamp((298 * c1 + 409 * e + 128) >> 8);
        rgb[4] = clamp((298 * c1 - 100 * d - 208 * e + 128) >> 8);
        rgb[5] = clamp((298 * c1 + 516 * d + 128) >> 8);
        rgb += 6;
    }
}

// V4L2 Control 操作
bool V4l2Camera::setControl(uint32_t id, int value) {
    struct v4l2_control ctrl;
    ctrl.id = id;
    ctrl.value = value;
    return xioctl(m_fd, VIDIOC_S_CTRL, &ctrl) == 0;
}

int V4l2Camera::getControl(uint32_t id) {
    struct v4l2_control ctrl;
    ctrl.id = id;
    if (xioctl(m_fd, VIDIOC_G_CTRL, &ctrl) < 0) return -1;
    return ctrl.value;
}

bool V4l2Camera::setExposure(int value) {
    return setControl(V4L2_CID_EXPOSURE, value);
}

bool V4l2Camera::setBrightness(int value) {
    return setControl(V4L2_CID_BRIGHTNESS, value);
}

bool V4l2Camera::setContrast(int value) {
    return setControl(V4L2_CID_CONTRAST, value);
}

bool V4l2Camera::setHFlip(bool enable) {
    return setControl(V4L2_CID_HFLIP, enable ? 1 : 0);
}

bool V4l2Camera::setVFlip(bool enable) {
    return setControl(V4L2_CID_VFLIP, enable ? 1 : 0);
}

// 采集线程主循环
void V4l2Camera::run() {
    if (!startStreaming()) return;
    m_running = true;

    // 分配 RGB 缓冲区
    std::vector<unsigned char> rgbBuf(m_width * m_height * 3);

    while (m_running) {
        void *data = nullptr;
        int size = 0;

        if (!grabFrame(&data, &size)) {
            if (m_running)
                emit errorOccurred("采集帧失败");
            continue;
        }

        // YUYV → RGB888
        yuyvToRgb888((const unsigned char *)data, rgbBuf.data(),
                      m_width, m_height);

        // 构造 QImage（深拷贝，因为 rgbBuf 会被下一帧覆盖）
        QImage image(rgbBuf.data(), m_width, m_height,
                     m_width * 3, QImage::Format_RGB888);
        emit frameReady(image.copy());
    }

    stopStreaming();
}
```

## 3.5 与 STM32MP157 的适配要点

### 像素格式

STM32MP157 的 DCMI 接口 + OV5640 通常支持：
- **YUYV (YUV422)**：最常见，上面代码默认使用
- **RGB565**：部分配置支持
- **NV12 (YUV420)**：部分配置支持

如果 YUYV 不支持，修改 setFormat() 中的 pixelformat，并相应修改颜色转换函数。

```bash
# 在开发板上查看实际支持的格式
v4l2-ctl --device=/dev/video0 --list-formats-ext
```

### 分辨率选择

根据开发板屏幕分辨率选择采集分辨率：
- 屏幕 800x480 → 采集 640x480 或 800x600
- 屏幕 480x272 → 采集 320x240 或 640x480（缩放显示）

### 帧率

```bash
# 查看支持的帧率
v4l2-ctl --device=/dev/video0 --list-formats-ext
# 设置帧率
v4l2-ctl --device=/dev/video0 --set-parm=30
```

在代码中设置帧率：

```cpp
struct v4l2_streamparm parm;
memset(&parm, 0, sizeof(parm));
parm.type = V4L2_BUF_TYPE_VIDEO_CAPTURE;
parm.parm.capture.timeperframe.numerator = 1;
parm.parm.capture.timeperframe.denominator = 30; // 30fps
xioctl(m_fd, VIDIOC_S_PARM, &parm);
```

## 3.6 单独测试采集模块

在集成到 Qt GUI 之前，先写一个简单的命令行测试程序：

```cpp
// test_capture.cpp
#include "v4l2_camera.h"
#include <QCoreApplication>
#include <QObject>
#include <QImage>
#include <QDebug>

int main(int argc, char *argv[]) {
    QCoreApplication app(argc, argv);

    V4l2Camera cam("/dev/video0", 640, 480);
    if (!cam.open()) {
        qDebug() << "打开摄像头失败";
        return -1;
    }

    int count = 0;
    QObject::connect(&cam, &V4l2Camera::frameReady, [&](const QImage &img) {
        if (count == 0) {
            img.save("/tmp/test_frame.jpg");
            qDebug() << "保存测试帧: /tmp/test_frame.jpg" << img.size();
        }
        count++;
        if (count >= 30) {
            qDebug() << "采集30帧完成";
            cam.stop();
            app.quit();
        }
    });

    cam.start();
    return app.exec();
}
```
