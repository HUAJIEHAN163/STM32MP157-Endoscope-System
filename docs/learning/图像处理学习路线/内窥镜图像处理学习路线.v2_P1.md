### 5. 图像分割（病变区域检测）

**为什么重要：** 内窥镜的核心价值是帮助医生发现病变，分割是第一步。

**颜色分割（最直接的方法）：**

```cpp
// 内窥镜中检测红色病变区域（如出血点、息肉）
Mat DetectRedLesion(const Mat &image) {
    Mat hsv;
    cvtColor(image, hsv, COLOR_BGR2HSV);

    // 红色在 HSV 中分两段
    Mat mask1, mask2, red_mask;
    inRange(hsv, Scalar(0, 70, 50), Scalar(10, 255, 255), mask1);
    inRange(hsv, Scalar(170, 70, 50), Scalar(180, 255, 255), mask2);
    red_mask = mask1 | mask2;

    // 形态学清理
    Mat kernel = getStructuringElement(MORPH_ELLIPSE, Size(5, 5));
    morphologyEx(red_mask, red_mask, MORPH_OPEN, kernel);
    morphologyEx(red_mask, red_mask, MORPH_CLOSE, kernel);

    return red_mask;
}
```

**轮廓检测（定位病变位置）：**

```cpp
// 找到病变区域的轮廓并标记
void MarkLesions(Mat &image, const Mat &mask) {
    vector<vector<Point>> contours;
    findContours(mask, contours, RETR_EXTERNAL, CHAIN_APPROX_SIMPLE);

    for (size_t i = 0; i < contours.size(); i++) {
        double area = contourArea(contours[i]);
        if (area < 100) continue;  // 过滤小噪点

        // 画矩形框标记病变
        Rect bbox = boundingRect(contours[i]);
        rectangle(image, bbox, Scalar(0, 255, 0), 2);

        // 计算面积占比
        char text[32];
        snprintf(text, sizeof(text), "%.1f%%",
                 area / (image.rows * image.cols) * 100);
        putText(image, text, bbox.tl(), FONT_HERSHEY_SIMPLEX,
                0.5, Scalar(0, 255, 0), 1);
    }
}
```

**分水岭分割（复杂场景）：**

```cpp
// 分水岭算法分割重叠的组织区域
Mat WatershedSegment(const Mat &image) {
    Mat gray, binary;
    cvtColor(image, gray, COLOR_BGR2GRAY);
    threshold(gray, binary, 0, 255, THRESH_BINARY_INV + THRESH_OTSU);

    // 距离变换找前景
    Mat dist;
    distanceTransform(binary, dist, DIST_L2, 5);
    normalize(dist, dist, 0, 1.0, NORM_MINMAX);

    Mat sure_fg;
    threshold(dist, sure_fg, 0.5, 255, THRESH_BINARY);
    sure_fg.convertTo(sure_fg, CV_8U);

    // 标记
    Mat markers;
    connectedComponents(sure_fg, markers);
    markers += 1;

    // 分水岭
    watershed(image, markers);
    return markers;
}
```

### 6. 视频流实时处理框架

**为什么重要：** 内窥镜是实时视频，不是静态图片。必须在 33ms 内处理完一帧。

**完整的实时处理管线：**

```cpp
class EndoscopeProcessor {
public:
    void Run(const string &video_source) {
        VideoCapture cap(video_source);
        Mat frame;
        int64 start_tick;

        while (cap.read(frame)) {
            start_tick = getTickCount();

            // 处理管线（顺序很重要）
            Mat result = frame.clone();
            Step1_WhiteBalance(result);
            Step2_Denoise(result);
            Step3_Enhance(result);
            Step4_DetectLesion(result);

            // 计算处理耗时
            double elapsed = (getTickCount() - start_tick)
                             / getTickFrequency() * 1000;

            char fps_text[32];
            snprintf(fps_text, sizeof(fps_text),
                     "%.1fms (%.0ffps)", elapsed, 1000.0 / elapsed);
            putText(result, fps_text, Point(10, 30),
                    FONT_HERSHEY_SIMPLEX, 0.8, Scalar(0, 255, 0), 2);

            imshow("Endoscope", result);
            if (waitKey(1) == 'q') break;
        }
    }

private:
    Ptr<CLAHE> clahe = createCLAHE(2.0, Size(8, 8));

    void Step1_WhiteBalance(Mat &frame) {
        Scalar mean_val = mean(frame);
        double avg = (mean_val[0] + mean_val[1] + mean_val[2]) / 3.0;
        frame.convertTo(frame, CV_32F);
        vector<Mat> ch;
        split(frame, ch);
        ch[0] *= avg / mean_val[0];
        ch[1] *= avg / mean_val[1];
        ch[2] *= avg / mean_val[2];
        merge(ch, frame);
        frame.convertTo(frame, CV_8U);
    }

    void Step2_Denoise(Mat &frame) {
        bilateralFilter(frame.clone(), frame, 5, 50, 50);
    }

    void Step3_Enhance(Mat &frame) {
        Mat lab;
        cvtColor(frame, lab, COLOR_BGR2Lab);
        vector<Mat> ch;
        split(lab, ch);
        clahe->apply(ch[0], ch[0]);
        merge(ch, lab);
        cvtColor(lab, frame, COLOR_Lab2BGR);
    }

    void Step4_DetectLesion(Mat &frame) {
        Mat hsv, mask1, mask2, red_mask;
        cvtColor(frame, hsv, COLOR_BGR2HSV);
        inRange(hsv, Scalar(0, 70, 50), Scalar(10, 255, 255), mask1);
        inRange(hsv, Scalar(170, 70, 50), Scalar(180, 255, 255), mask2);
        red_mask = mask1 | mask2;

        Mat kernel = getStructuringElement(MORPH_ELLIPSE, Size(5, 5));
        morphologyEx(red_mask, red_mask, MORPH_CLOSE, kernel);

        vector<vector<Point>> contours;
        findContours(red_mask, contours, RETR_EXTERNAL,
                     CHAIN_APPROX_SIMPLE);
        for (auto &c : contours) {
            if (contourArea(c) > 200) {
                Rect r = boundingRect(c);
                rectangle(frame, r, Scalar(0, 0, 255), 2);
                putText(frame, "Lesion?", r.tl(),
                        FONT_HERSHEY_SIMPLEX, 0.5, Scalar(0, 0, 255), 1);
            }
        }
    }
};
```

### 7. 性能优化（嵌入式平台关键）

**为什么重要：** STM32MP157 的 A7 核性能有限，必须优化才能达到实时。

**多线程并行处理：**

```cpp
#include <thread>
#include <mutex>
#include <condition_variable>
#include <queue>

// 生产者-消费者模型：采集线程 + 处理线程
class PipelineProcessor {
    queue<Mat> frame_queue;
    mutex mtx;
    condition_variable cv;
    bool running = true;

public:
    // 采集线程：只负责读取帧
    void CaptureThread(const string &source) {
        VideoCapture cap(source);
        Mat frame;
        while (running && cap.read(frame)) {
            lock_guard<mutex> lock(mtx);
            if (frame_queue.size() < 3)  // 最多缓存 3 帧
                frame_queue.push(frame.clone());
            cv.notify_one();
        }
    }

    // 处理线程：只负责算法处理
    void ProcessThread() {
        while (running) {
            Mat frame;
            {
                unique_lock<mutex> lock(mtx);
                cv.wait(lock, [this]{ return !frame_queue.empty() || !running; });
                if (!running) break;
                frame = frame_queue.front();
                frame_queue.pop();
            }
            // 图像处理（不持有锁，不阻塞采集）
            ProcessFrame(frame);
            imshow("Result", frame);
            waitKey(1);
        }
    }

    void ProcessFrame(Mat &frame) {
        // CLAHE + 去噪 + 检测
    }
};
```

**NEON 指令加速（ARM 专用）：**

```cpp
// STM32MP157 的 A7 支持 NEON SIMD 指令
// OpenCV 编译时开启 NEON 支持即可自动加速
// cmake 编译选项：
// -DENABLE_NEON=ON -DCMAKE_C_FLAGS="-mfpu=neon"

// 手动 NEON 优化示例（亮度调节）
#include <arm_neon.h>

void AdjustBrightness_NEON(uint8_t *data, int size, int delta) {
    uint8x16_t vdelta = vdupq_n_u8(abs(delta));
    int i = 0;
    for (; i + 16 <= size; i += 16) {
        uint8x16_t pixels = vld1q_u8(data + i);
        if (delta > 0)
            pixels = vqaddq_u8(pixels, vdelta);  // 饱和加法
        else
            pixels = vqsubq_u8(pixels, vdelta);  // 饱和减法
        vst1q_u8(data + i, pixels);
    }
    // 处理剩余像素
    for (; i < size; i++) {
        int val = data[i] + delta;
        data[i] = (val > 255) ? 255 : (val < 0) ? 0 : val;
    }
}
```

### 8. 图像质量评估

**为什么重要：** 需要自动判断图像质量，质量差时提醒医生或自动调整参数。

```cpp
// 1. 清晰度评估（拉普拉斯方差法）
double CalcSharpness(const Mat &image) {
    Mat gray, lap;
    cvtColor(image, gray, COLOR_BGR2GRAY);
    Laplacian(gray, lap, CV_64F);
    Scalar mean, stddev;
    meanStdDev(lap, mean, stddev);
    return stddev[0] * stddev[0];
    // > 100: 清晰  |  < 50: 模糊，需要提醒
}

// 2. 亮度评估
double CalcBrightness(const Mat &image) {
    Mat gray;
    cvtColor(image, gray, COLOR_BGR2GRAY);
    return mean(gray)[0];
    // 理想范围：80~180
    // < 60: 欠曝，需要增加补光
    // > 200: 过曝，需要降低补光
}

// 3. 综合质量评估
struct ImageQuality {
    double sharpness;
    double brightness;
    bool is_acceptable;
};

ImageQuality EvaluateFrame(const Mat &frame) {
    ImageQuality q;
    q.sharpness = CalcSharpness(frame);
    q.brightness = CalcBrightness(frame);
    q.is_acceptable = (q.sharpness > 50) && (q.brightness > 60)
                      && (q.brightness < 200);
    return q;
}
```
