### 1. 数字图像基础（3天）

**为什么是第一优先级：** 所有图像算法都建立在这些概念之上。

**图像的本质：**

```
一张 1920×1080 的 RGB 图像：
- 像素总数：1920 × 1080 = 2,073,600 个像素
- 每个像素：3 个通道（R, G, B），每通道 8 位（0~255）
- 内存占用：2,073,600 × 3 = 6,220,800 字节 ≈ 6MB
- 30fps 视频带宽：6MB × 30 = 180MB/s
```

**色彩空间（内窥镜开发必须精通）：**

| 色彩空间 | 用途 | 内窥镜场景 |
|---------|------|-----------|
| RGB | 显示输出 | 最终显示到屏幕 |
| YUV/YCbCr | 摄像头输出、视频编码 | 摄像头原始数据格式 |
| HSV | 颜色分割 | 检测红色病变区域 |
| Lab | 感知均匀的颜色空间 | 颜色校正、白平衡 |
| Gray | 灰度图 | 边缘检测、特征提取 |

```cpp
#include <opencv2/opencv.hpp>
using namespace cv;

// 内窥镜中最常用的色彩空间转换
Mat image = imread("endoscope.jpg");

// 1. BGR → YUV（理解摄像头输出格式）
Mat yuv;
cvtColor(image, yuv, COLOR_BGR2YUV);

// 2. BGR → HSV（用于检测红色病变）
Mat hsv;
cvtColor(image, hsv, COLOR_BGR2HSV);
// 红色在 HSV 中：H=0~10 或 170~180
Mat red_mask;
inRange(hsv, Scalar(0, 100, 100), Scalar(10, 255, 255), red_mask);

// 3. BGR → Lab（用于颜色校正）
Mat lab;
cvtColor(image, lab, COLOR_BGR2Lab);
```

**像素级操作（必须熟练）：**

```cpp
// 遍历图像像素（理解内存布局）
void AdjustBrightness(Mat &image, int delta) {
    for (int y = 0; y < image.rows; y++) {
        uchar *row = image.ptr<uchar>(y);  // 获取第 y 行指针
        for (int x = 0; x < image.cols * 3; x++) {
            int val = row[x] + delta;
            row[x] = (val > 255) ? 255 : (val < 0) ? 0 : val;
        }
    }
}

// 更高效的方式：使用 LUT（查找表）
void AdjustBrightnessLUT(Mat &image, int delta) {
    Mat lut(1, 256, CV_8UC1);
    for (int i = 0; i < 256; i++) {
        int val = i + delta;
        lut.at<uchar>(i) = saturate_cast<uchar>(val);
    }
    LUT(image, lut, image);  // 硬件级别的速度
}
```

**滤波（内窥镜图像去噪核心）：**

```cpp
// 1. 高斯滤波（最基础的去噪）
Mat blurred;
GaussianBlur(image, blurred, Size(5, 5), 1.5);

// 2. 双边滤波（去噪但保留边缘，内窥镜首选）
Mat denoised;
bilateralFilter(image, denoised, 9, 75, 75);

// 3. 非局部均值去噪（效果最好但最慢）
Mat nlm_denoised;
fastNlMeansDenoisingColored(image, nlm_denoised, 10, 10, 7, 21);
```

**边缘检测（识别组织边界）：**

```cpp
Mat gray, edges;
cvtColor(image, gray, COLOR_BGR2GRAY);

// Canny 边缘检测（最常用）
Canny(gray, edges, 50, 150);

// Sobel 梯度（方向性边缘）
Mat grad_x, grad_y;
Sobel(gray, grad_x, CV_16S, 1, 0);  // X 方向梯度
Sobel(gray, grad_y, CV_16S, 0, 1);  // Y 方向梯度
```

**形态学操作（清理噪声、填充空洞）：**

```cpp
Mat kernel = getStructuringElement(MORPH_ELLIPSE, Size(5, 5));

// 开运算：去除小噪点
morphologyEx(binary, binary, MORPH_OPEN, kernel);

// 闭运算：填充小空洞
morphologyEx(binary, binary, MORPH_CLOSE, kernel);

// 膨胀 + 腐蚀（手动控制）
dilate(binary, binary, kernel, Point(-1,-1), 2);
erode(binary, binary, kernel, Point(-1,-1), 1);
```

### 3. 内窥镜图像增强（3天）

**为什么是 P0：** 这是内窥镜最基本也最核心的功能，每一帧都需要增强。

**对比度增强（CLAHE，内窥镜标配算法）：**

```cpp
// CLAHE：自适应直方图均衡化
// 内窥镜图像光照不均匀，普通均衡化效果差，CLAHE 分块处理
Mat enhanced;
Mat lab;
cvtColor(image, lab, COLOR_BGR2Lab);

vector<Mat> channels;
split(lab, channels);

// 只对亮度通道（L）做 CLAHE
Ptr<CLAHE> clahe = createCLAHE(2.0, Size(8, 8));
clahe->apply(channels[0], channels[0]);

merge(channels, lab);
cvtColor(lab, enhanced, COLOR_Lab2BGR);
```

**白平衡校正（补光灯色温补偿）：**

```cpp
// 灰度世界假设法（简单有效）
Mat WhiteBalance_GrayWorld(const Mat &image) {
    Scalar mean_val = mean(image);
    double avg = (mean_val[0] + mean_val[1] + mean_val[2]) / 3.0;

    Mat result;
    image.convertTo(result, CV_32F);

    vector<Mat> channels;
    split(result, channels);
    channels[0] *= avg / mean_val[0];  // B
    channels[1] *= avg / mean_val[1];  // G
    channels[2] *= avg / mean_val[2];  // R
    merge(channels, result);

    result.convertTo(result, CV_8U);
    return result;
}
```

**去雾/去雾化（内窥镜体液干扰）：**

```cpp
// 暗通道先验去雾（内窥镜中体液、水雾导致图像模糊）
Mat DarkChannel(const Mat &image, int patch_size) {
    Mat dark(image.rows, image.cols, CV_8UC1);
    int half = patch_size / 2;

    for (int y = 0; y < image.rows; y++) {
        for (int x = 0; x < image.cols; x++) {
            uchar min_val = 255;
            for (int dy = -half; dy <= half; dy++) {
                for (int dx = -half; dx <= half; dx++) {
                    int ny = min(max(y + dy, 0), image.rows - 1);
                    int nx = min(max(x + dx, 0), image.cols - 1);
                    Vec3b pixel = image.at<Vec3b>(ny, nx);
                    min_val = min({min_val, pixel[0], pixel[1], pixel[2]});
                }
            }
            dark.at<uchar>(y, x) = min_val;
        }
    }
    return dark;
}
```

### 4. V4L2 摄像头编程（3天）

**为什么是 P0：** 在 Linux 上采集摄像头数据是图像处理的第一步。

```c
// V4L2 是 Linux 下摄像头的标准接口
#include <linux/videodev2.h>
#include <sys/ioctl.h>
#include <sys/mman.h>

int Camera_Open(const char *dev) {
    int fd = open(dev, O_RDWR);

    // 设置格式
    struct v4l2_format fmt = {0};
    fmt.type = V4L2_BUF_TYPE_VIDEO_CAPTURE;
    fmt.fmt.pix.width = 1920;
    fmt.fmt.pix.height = 1080;
    fmt.fmt.pix.pixelformat = V4L2_PIX_FMT_YUYV;
    ioctl(fd, VIDIOC_S_FMT, &fmt);

    // 申请缓冲区
    struct v4l2_requestbuffers req = {0};
    req.count = 4;
    req.type = V4L2_BUF_TYPE_VIDEO_CAPTURE;
    req.memory = V4L2_MEMORY_MMAP;
    ioctl(fd, VIDIOC_REQBUFS, &req);

    return fd;
}
```
