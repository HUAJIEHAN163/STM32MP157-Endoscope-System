### 确认摄像头识别

```bash
# 在开发板上执行
# 1. 插入 USB 摄像头
# 2. 查看内核日志
dmesg | tail -20

# usb 1-1: new high-speed USB device
# uvcvideo: Found UVC 1.00 device USB Camera

# 3. 确认视频设备节点

# 4. 查看摄像头支持的格式
v4l2-ctl --device=/dev/video0 --list-formats-ext

#   Size: 1280x720  - Interval: 1/30

```

## Step 3.2：V4L2 采集图像

### V4L2 采集封装

```bash
mkdir -p ~/mp157/endoscope/src ~/mp157/endoscope/include
cd ~/mp157/endoscope
```

**include/camera.h：**

```c
#ifndef CAMERA_H
#define CAMERA_H

#include <stdint.h>

typedef struct {
    int fd;
    uint32_t width;
    uint32_t height;
    uint8_t *buffers[4];
    uint32_t buf_lengths[4];
    int buf_count;
} Camera_t;

int camera_open(Camera_t *cam, const char *dev,
                uint32_t width, uint32_t height);
int camera_start(Camera_t *cam);
int camera_grab_frame(Camera_t *cam, uint8_t **data, uint32_t *length);
int camera_release_frame(Camera_t *cam, int buf_index);
void camera_stop(Camera_t *cam);
void camera_close(Camera_t *cam);

#endif
```

**src/camera.c：**

```c
#include "camera.h"
#include <stdio.h>
#include <string.h>
#include <fcntl.h>
#include <unistd.h>
#include <sys/ioctl.h>
#include <sys/mman.h>
#include <linux/videodev2.h>

static int last_buf_index = -1;

int camera_open(Camera_t *cam, const char *dev,
                uint32_t width, uint32_t height) {
    memset(cam, 0, sizeof(Camera_t));
    cam->width = width;
    cam->height = height;

    // 打开设备
    cam->fd = open(dev, O_RDWR);
    if (cam->fd < 0) {
        perror("open camera");
        return -1;
    }

    // 设置格式
    struct v4l2_format fmt = {0};
    fmt.type = V4L2_BUF_TYPE_VIDEO_CAPTURE;
    fmt.fmt.pix.width = width;
    fmt.fmt.pix.height = height;
    fmt.fmt.pix.pixelformat = V4L2_PIX_FMT_YUYV;
    fmt.fmt.pix.field = V4L2_FIELD_NONE;
    if (ioctl(cam->fd, VIDIOC_S_FMT, &fmt) < 0) {
        perror("set format");
        close(cam->fd);
        return -1;
    }

    // 申请缓冲区
    struct v4l2_requestbuffers req = {0};
    req.count = 4;
    req.type = V4L2_BUF_TYPE_VIDEO_CAPTURE;
    req.memory = V4L2_MEMORY_MMAP;
    if (ioctl(cam->fd, VIDIOC_REQBUFS, &req) < 0) {
        perror("request buffers");
        close(cam->fd);
        return -1;
    }
    cam->buf_count = req.count;

    // 映射缓冲区到用户空间
    for (int i = 0; i < cam->buf_count; i++) {
        struct v4l2_buffer buf = {0};
        buf.type = V4L2_BUF_TYPE_VIDEO_CAPTURE;
        buf.memory = V4L2_MEMORY_MMAP;
        buf.index = i;
        ioctl(cam->fd, VIDIOC_QUERYBUF, &buf);

        cam->buf_lengths[i] = buf.length;
        cam->buffers[i] = mmap(NULL, buf.length,
                               PROT_READ | PROT_WRITE, MAP_SHARED,
                               cam->fd, buf.m.offset);
    }

    printf("Camera opened: %dx%d, %d buffers\n",
           width, height, cam->buf_count);
    return 0;
}

int camera_start(Camera_t *cam) {
    // 将所有缓冲区入队
    for (int i = 0; i < cam->buf_count; i++) {
        struct v4l2_buffer buf = {0};
        buf.type = V4L2_BUF_TYPE_VIDEO_CAPTURE;
        buf.memory = V4L2_MEMORY_MMAP;
        buf.index = i;
        ioctl(cam->fd, VIDIOC_QBUF, &buf);
    }

    // 开始采集
    enum v4l2_buf_type type = V4L2_BUF_TYPE_VIDEO_CAPTURE;
    if (ioctl(cam->fd, VIDIOC_STREAMON, &type) < 0) {
        perror("stream on");
        return -1;
    }
    printf("Camera streaming started\n");
    return 0;
}

int camera_grab_frame(Camera_t *cam, uint8_t **data, uint32_t *length) {
    struct v4l2_buffer buf = {0};
    buf.type = V4L2_BUF_TYPE_VIDEO_CAPTURE;
    buf.memory = V4L2_MEMORY_MMAP;

    if (ioctl(cam->fd, VIDIOC_DQBUF, &buf) < 0) {
        perror("dequeue buffer");
        return -1;
    }

    *data = cam->buffers[buf.index];
    *length = buf.bytesused;
    last_buf_index = buf.index;
    return buf.index;
}

int camera_release_frame(Camera_t *cam, int buf_index) {
    struct v4l2_buffer buf = {0};
    buf.type = V4L2_BUF_TYPE_VIDEO_CAPTURE;
    buf.memory = V4L2_MEMORY_MMAP;
    buf.index = buf_index;
    return ioctl(cam->fd, VIDIOC_QBUF, &buf);
}

void camera_stop(Camera_t *cam) {
    enum v4l2_buf_type type = V4L2_BUF_TYPE_VIDEO_CAPTURE;
    ioctl(cam->fd, VIDIOC_STREAMOFF, &type);
}

void camera_close(Camera_t *cam) {
    camera_stop(cam);
    for (int i = 0; i < cam->buf_count; i++) {
        munmap(cam->buffers[i], cam->buf_lengths[i]);
    }
    close(cam->fd);
    printf("Camera closed\n");
}
```

### 纯 C 测试（不依赖 OpenCV）

**src/test_camera.c：**

```c
#include "camera.h"
#include <stdio.h>

int main() {
    Camera_t cam;
    if (camera_open(&cam, "/dev/video0", 640, 480) < 0)
        return 1;

    camera_start(&cam);

    // 抓 10 帧
    for (int i = 0; i < 10; i++) {
        uint8_t *data;
        uint32_t length;
        int idx = camera_grab_frame(&cam, &data, &length);
        if (idx >= 0) {
            printf("Frame %d: %u bytes\n", i, length);
            camera_release_frame(&cam, idx);
        }
    }

    camera_close(&cam);
    printf("Camera test passed!\n");
    return 0;
}
```

```bash
# 编译测试

    -Iinclude -Wall -O2

# 部署并运行

ssh root@192.168.1.100 "./test_camera"
# 输出：
# Camera opened: 640x480, 4 buffers
# Camera streaming started
# Frame 0: 614400 bytes
# Frame 1: 614400 bytes
# ...
# Camera test passed!
```

## Step 3.3：OpenCV 交叉编译

### 方法 1：使用开发板自带的 OpenCV（推荐）

```bash
# 野火的 Linux 镜像通常已预装 OpenCV
# 在开发板上检查
ssh root@192.168.1.100
pkg-config --modversion opencv4
# 如果输出版本号（如 4.5.1），说明已安装

# 查看头文件和库路径
pkg-config --cflags --libs opencv4
```

### 方法 2：手动交叉编译 OpenCV

```bash
# 在 Ubuntu 虚拟机上执行

# 1. 下载 OpenCV 源码
cd ~/mp157
git clone https://github.com/opencv/opencv.git -b 4.8.0 --depth 1

# 2. 创建编译目录
mkdir opencv_build && cd opencv_build

# 3. CMake 配置（交叉编译）
cmake ../opencv \
    -DCMAKE_TOOLCHAIN_FILE=../opencv/platforms/linux/arm-gnueabi.toolchain.cmake \
    -DCMAKE_INSTALL_PREFIX=~/mp157/opencv_install \
    -DBUILD_SHARED_LIBS=ON \
    -DBUILD_TESTS=OFF \
    -DBUILD_PERF_TESTS=OFF \
    -DBUILD_EXAMPLES=OFF \
    -DBUILD_opencv_python3=OFF \
    -DWITH_GTK=OFF \
    -DWITH_FFMPEG=ON \
    -DENABLE_NEON=ON \
    -DCMAKE_C_FLAGS="-march=armv7-a -mfpu=neon" \
    -DCMAKE_CXX_FLAGS="-march=armv7-a -mfpu=neon"

# 4. 编译（耗时较长，约 30-60 分钟）
make -j$(nproc)
make install

# 5. 将库文件部署到开发板
scp -r ~/mp157/opencv_install/lib/* root@192.168.1.100:/usr/lib/
scp -r ~/mp157/opencv_install/include/* root@192.168.1.100:/usr/include/
```
