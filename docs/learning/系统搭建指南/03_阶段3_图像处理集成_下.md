## Step 3.4：图像增强管线

### 完整的图像处理器

**include/image_processor.h：**

```cpp
#ifndef IMAGE_PROCESSOR_H
#define IMAGE_PROCESSOR_H

#include <opencv2/opencv.hpp>

class ImageProcessor {
public:
    ImageProcessor();

    // 完整处理管线：输入原始帧，输出增强后的帧
    cv::Mat Process(const cv::Mat &frame);

    // 获取当前帧的平均亮度（用于控制补光灯）
    double GetBrightness() const;

private:
    cv::Ptr<cv::CLAHE> clahe_;
    double current_brightness_;

    void WhiteBalance(cv::Mat &frame);
    void Denoise(cv::Mat &frame);
    void EnhanceContrast(cv::Mat &frame);
};

#endif
```

**src/image_processor.cpp：**

```cpp
#include "image_processor.h"

ImageProcessor::ImageProcessor()
    : current_brightness_(0) {
    clahe_ = cv::createCLAHE(2.0, cv::Size(8, 8));
}

cv::Mat ImageProcessor::Process(const cv::Mat &frame) {
    cv::Mat result = frame.clone();

    // 处理管线（顺序重要）
    WhiteBalance(result);
    Denoise(result);
    EnhanceContrast(result);

    // 计算亮度
    cv::Mat gray;
    cv::cvtColor(result, gray, cv::COLOR_BGR2GRAY);
    current_brightness_ = cv::mean(gray)[0];

    return result;
}

double ImageProcessor::GetBrightness() const {
    return current_brightness_;
}

void ImageProcessor::WhiteBalance(cv::Mat &frame) {
    // 灰度世界假设法
    cv::Scalar mean_val = cv::mean(frame);
    double avg = (mean_val[0] + mean_val[1] + mean_val[2]) / 3.0;

    std::vector<cv::Mat> channels;
    cv::split(frame, channels);
    channels[0] *= avg / mean_val[0];  // B
    channels[1] *= avg / mean_val[1];  // G
    channels[2] *= avg / mean_val[2];  // R
    cv::merge(channels, frame);
}

void ImageProcessor::Denoise(cv::Mat &frame) {
    cv::Mat temp;
    cv::bilateralFilter(frame, temp, 5, 50, 50);
    frame = temp;
}

void ImageProcessor::EnhanceContrast(cv::Mat &frame) {
    cv::Mat lab;
    cv::cvtColor(frame, lab, cv::COLOR_BGR2Lab);

    std::vector<cv::Mat> channels;
    cv::split(lab, channels);
    clahe_->apply(channels[0], channels[0]);
    cv::merge(channels, lab);

    cv::cvtColor(lab, frame, cv::COLOR_Lab2BGR);
}
```

```cpp
};

```

**src/lesion_detector.cpp：**

```cpp
#include "lesion_detector.h"

std::vector<Lesion> LesionDetector::Detect(cv::Mat &frame) {
    std::vector<Lesion> results;

    // HSV 颜色分割：检测红色区域
    cv::Mat hsv, mask1, mask2, red_mask;
    cv::cvtColor(frame, hsv, cv::COLOR_BGR2HSV);
    cv::inRange(hsv, cv::Scalar(0, 70, 50),
                cv::Scalar(10, 255, 255), mask1);
    cv::inRange(hsv, cv::Scalar(170, 70, 50),
                cv::Scalar(180, 255, 255), mask2);
    red_mask = mask1 | mask2;

    // 形态学清理
    cv::Mat kernel = cv::getStructuringElement(
        cv::MORPH_ELLIPSE, cv::Size(5, 5));
    cv::morphologyEx(red_mask, red_mask, cv::MORPH_OPEN, kernel);
    cv::morphologyEx(red_mask, red_mask, cv::MORPH_CLOSE, kernel);

    // 找轮廓
    std::vector<std::vector<cv::Point>> contours;
    cv::findContours(red_mask, contours, cv::RETR_EXTERNAL,
                     cv::CHAIN_APPROX_SIMPLE);

    double total_area = frame.rows * frame.cols;

    for (auto &contour : contours) {
        double area = cv::contourArea(contour);
        if (area < 200) continue;  // 过滤噪点

        Lesion l;
        l.bbox = cv::boundingRect(contour);
        l.area = area;
        l.area_ratio = area / total_area * 100.0;
        results.push_back(l);

        // 在画面上标记
        cv::rectangle(frame, l.bbox, cv::Scalar(0, 255, 0), 2);
        char text[32];
        snprintf(text, sizeof(text), "%.1f%%", l.area_ratio);
        cv::putText(frame, text, l.bbox.tl(),
                    cv::FONT_HERSHEY_SIMPLEX, 0.5,
                    cv::Scalar(0, 255, 0), 1);
    }

    return results;
}
```

## Step 3.6：图像亮度 → RPMsg → M4 调节补光灯

### RPMsg 通信封装（C++ 版本）

**include/rpmsg_comm.h：**

```cpp
#ifndef RPMSG_COMM_H
#define RPMSG_COMM_H

#include <string>

class RPMsgComm {
public:
    bool Open(const std::string &dev = "/dev/ttyRPMSG0");
    void Close();
    bool Send(const std::string &cmd);
    std::string Recv(int timeout_ms = 100);

    // 便捷方法
    void SetBrightness(int percent);
    float GetTemperature();

private:
    int fd_ = -1;
};

#endif
```

**src/rpmsg_comm.cpp：**

```cpp
#include "rpmsg_comm.h"
#include <fcntl.h>
#include <unistd.h>
#include <poll.h>
#include <cstdio>
#include <cstring>
#include <cstdlib>

bool RPMsgComm::Open(const std::string &dev) {
    fd_ = open(dev.c_str(), O_RDWR | O_NOCTTY);
    if (fd_ < 0) {
        perror("RPMsg open");
        return false;
    }
    return true;
}

void RPMsgComm::Close() {
    if (fd_ >= 0) {
        close(fd_);
        fd_ = -1;
    }
}

bool RPMsgComm::Send(const std::string &cmd) {
    if (fd_ < 0) return false;
    return write(fd_, cmd.c_str(), cmd.size()) == (ssize_t)cmd.size();
}

std::string RPMsgComm::Recv(int timeout_ms) {
    if (fd_ < 0) return "";
    struct pollfd pfd = { .fd = fd_, .events = POLLIN, .revents = 0 };
    if (poll(&pfd, 1, timeout_ms) > 0 && (pfd.revents & POLLIN)) {
        char buf[128];
        int n = read(fd_, buf, sizeof(buf) - 1);
        if (n > 0) {
            buf[n] = '\0';
            return std::string(buf);
        }
    }
    return "";
}

void RPMsgComm::SetBrightness(int percent) {
    char cmd[32];
    snprintf(cmd, sizeof(cmd), "bright %d", percent);
    Send(cmd);
}

float RPMsgComm::GetTemperature() {
    Send("temp");
    std::string reply = Recv(200);
    // 解析 "Temp: 28.5 C\r\n"
    float temp = 0;
    sscanf(reply.c_str(), "Temp: %f", &temp);
    return temp;
}
```

### 自动亮度调节逻辑

```cpp
// 在主循环中调用
void AutoAdjustBrightness(RPMsgComm &rpmsg, double image_brightness) {
    // 目标亮度范围：100~160（8位灰度值）
    const double TARGET_LOW = 100.0;
    const double TARGET_HIGH = 160.0;
    const double TARGET_MID = 130.0;

    static int current_led_brightness = 50;  // 初始 50%

    if (image_brightness < TARGET_LOW) {
        // 图像太暗，增加补光
        current_led_brightness += 5;
        if (current_led_brightness > 100) current_led_brightness = 100;
        rpmsg.SetBrightness(current_led_brightness);
    } else if (image_brightness > TARGET_HIGH) {
        // 图像太亮，降低补光
        current_led_brightness -= 5;
        if (current_led_brightness < 0) current_led_brightness = 0;
        rpmsg.SetBrightness(current_led_brightness);
    }
    // 在目标范围内不调整，避免频繁抖动
}
```

## Step 3.7：完整系统联调

### 主程序（整合所有模块）

**src/main.cpp：**

```cpp
#include <cstdio>
#include <csignal>
#include <opencv2/opencv.hpp>

extern "C" {
#include "camera.h"
}
#include "image_processor.h"
#include "lesion_detector.h"
#include "rpmsg_comm.h"

static volatile bool running = true;

void signal_handler(int sig) {
    (void)sig;
    running = false;
}

// YUYV → BGR 转换
cv::Mat YUYVtoBGR(uint8_t *data, int width, int height) {
    cv::Mat yuyv(height, width, CV_8UC2, data);
    cv::Mat bgr;
    cv::cvtColor(yuyv, bgr, cv::COLOR_YUV2BGR_YUYV);
    return bgr;
}

int main() {
    signal(SIGINT, signal_handler);

    // 1. 打开摄像头
    Camera_t cam;
    if (camera_open(&cam, "/dev/video0", 640, 480) < 0) {
        printf("Failed to open camera\n");
        return 1;
    }
    camera_start(&cam);

    // 2. 打开 RPMsg 通信
    RPMsgComm rpmsg;
    if (!rpmsg.Open()) {
        printf("Failed to open RPMsg (M4 not running?)\n");
        printf("Continuing without M4 control...\n");
    }

    // 3. 初始化图像处理器
    ImageProcessor processor;
    LesionDetector detector;

    // 4. 设置初始补光灯亮度
    rpmsg.SetBrightness(50);

    printf("=== Endoscope System Started ===\n");
    printf("Press Ctrl+C to quit\n\n");

    int frame_count = 0;
    int64_t start_time = cv::getTickCount();

    while (running) {
        // 采集一帧
        uint8_t *raw_data;
        uint32_t raw_length;
        int buf_idx = camera_grab_frame(&cam, &raw_data, &raw_length);
        if (buf_idx < 0) continue;

        // YUYV → BGR
        cv::Mat frame = YUYVtoBGR(raw_data, cam.width, cam.height);
        camera_release_frame(&cam, buf_idx);

        // 图像增强
        cv::Mat enhanced = processor.Process(frame);

        // 病变检测
        std::vector<Lesion> lesions = detector.Detect(enhanced);

        // 自动调节补光灯
        AutoAdjustBrightness(rpmsg, processor.GetBrightness());

        // 显示信息
        frame_count++;
        double elapsed = (cv::getTickCount() - start_time)
                         / cv::getTickFrequency();
        double fps = frame_count / elapsed;

        char info[128];
        snprintf(info, sizeof(info),
                 "FPS:%.1f Bright:%.0f Lesions:%zu",
                 fps, processor.GetBrightness(), lesions.size());
        cv::putText(enhanced, info, cv::Point(10, 25),
                    cv::FONT_HERSHEY_SIMPLEX, 0.6,
                    cv::Scalar(0, 255, 0), 1);

        // 安全检查：每 30 帧查一次温度
        if (frame_count % 30 == 0) {
            float temp = rpmsg.GetTemperature();
            if (temp > 0) {
                printf("Temp: %.1f C | FPS: %.1f | Lesions: %zu\n",
                       temp, fps, lesions.size());
            }
        }

        // 显示画面（如果有显示器）
        // cv::imshow("Endoscope", enhanced);
        // if (cv::waitKey(1) == 'q') break;

        // 无显示器时保存截图
        if (frame_count == 1) {
            cv::imwrite("/tmp/endoscope_frame.jpg", enhanced);
            printf("First frame saved to /tmp/endoscope_frame.jpg\n");
        }
    }

    printf("\n=== Shutting down ===\n");
    rpmsg.SetBrightness(0);
    rpmsg.Close();
    camera_close(&cam);
    return 0;
}
```

## 完整数据流回顾

```
USB 摄像头
    │ YUYV 640×480 @30fps
    ▼
V4L2 采集 (camera.c)
    │ 原始帧数据
    ▼
YUYV → BGR 转换
    │ BGR Mat
    ▼
图像增强 (image_processor.cpp)
    ├─ 白平衡校正
    ├─ 双边滤波去噪
    ├─ CLAHE 对比度增强
    │ 增强后的 BGR Mat + 亮度值
    ▼
病变检测 (lesion_detector.cpp)
    ├─ HSV 颜色分割
    ├─ 形态学清理
    ├─ 轮廓检测 + 标记
    │ 标记后的画面 + 病变列表
    ▼
补光灯控制
    ├─ 亮度值 → RPMsg → M4 → PWM 调节 LED
    │
    ▼
显示 / 保存
```
