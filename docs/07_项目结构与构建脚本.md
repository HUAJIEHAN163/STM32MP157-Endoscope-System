## 7.2 CMakeLists.txt

```cmake
cmake_minimum_required(VERSION 3.10)
project(EndoscopeDemo LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 11)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_AUTOMOC ON)  # Qt MOC 自动处理

# ============================================================
# 查找依赖
# ============================================================

# Qt5
find_package(Qt5 REQUIRED COMPONENTS Core Gui Widgets)

# OpenCV
find_package(OpenCV REQUIRED COMPONENTS core imgproc imgcodecs videoio)

# ============================================================
# 主程序
# ============================================================
set(SOURCES
    src/main.cpp
    src/mainwindow.cpp
    src/v4l2_camera.cpp
    src/image_processor.cpp
    src/util.cpp
)

set(HEADERS
    src/mainwindow.h
    src/v4l2_camera.h
    src/image_processor.h
    src/util.h
)

add_executable(endoscope_demo ${SOURCES} ${HEADERS})

target_include_directories(endoscope_demo PRIVATE
    ${CMAKE_CURRENT_SOURCE_DIR}/src
    ${OpenCV_INCLUDE_DIRS}
)

target_link_libraries(endoscope_demo
    Qt5::Core Qt5::Gui Qt5::Widgets
    ${OpenCV_LIBS}
    pthread
)

# ============================================================
# 测试工具
# ============================================================
add_executable(test_capture tools/test_capture.cpp
    src/v4l2_camera.cpp src/v4l2_camera.h)
target_include_directories(test_capture PRIVATE src/)
target_link_libraries(test_capture Qt5::Core pthread)

# ============================================================
# 标定工具（仅在宿主机编译）
# ============================================================
if(NOT CMAKE_CROSSCOMPILING)
    add_executable(calibrate_tool tools/calibrate_tool.cpp)
    target_link_libraries(calibrate_tool ${OpenCV_LIBS})
endif()
```

### scripts/build.sh

```bash
#!/bin/bash
set -e

SCRIPT_DIR="$(cd "$(dirname "$0")" && pwd)"
PROJECT_DIR="$(dirname "$SCRIPT_DIR")"
BUILD_DIR="$PROJECT_DIR/build_arm"

# 加载环境
source "$SCRIPT_DIR/setup_env.sh"

# 创建构建目录
mkdir -p "$BUILD_DIR"
cd "$BUILD_DIR"

# CMake 配置
cmake "$PROJECT_DIR" \
    -DCMAKE_BUILD_TYPE=Release \
    -DCMAKE_TOOLCHAIN_FILE="$PROJECT_DIR/scripts/arm_toolchain.cmake" \
    -DQt5_DIR=/opt/qt5_arm/lib/cmake/Qt5 \
    -DOpenCV_DIR=/opt/opencv_arm/lib/cmake/opencv4

# 编译
make -j$(nproc)

echo ""
echo "========================================="
echo "编译完成！"
echo "可执行文件: $BUILD_DIR/endoscope_demo"
echo "测试工具:   $BUILD_DIR/test_capture"
echo "========================================="
```

### scripts/arm_toolchain.cmake

```cmake
# CMake 交叉编译工具链文件
set(CMAKE_SYSTEM_NAME Linux)
set(CMAKE_SYSTEM_PROCESSOR arm)

# 根据实际路径修改
set(TOOLCHAIN_PREFIX arm-linux-gnueabihf-)

set(CMAKE_C_COMPILER   ${TOOLCHAIN_PREFIX}gcc)
set(CMAKE_CXX_COMPILER ${TOOLCHAIN_PREFIX}g++)

# sysroot（如果使用 SDK）
# set(CMAKE_SYSROOT /opt/st/stm32mp1/sysroots/cortexa7t2hf-neon-vfpv4-ostl-linux-gnueabi)

set(CMAKE_FIND_ROOT_PATH_MODE_PROGRAM NEVER)
set(CMAKE_FIND_ROOT_PATH_MODE_LIBRARY ONLY)
set(CMAKE_FIND_ROOT_PATH_MODE_INCLUDE ONLY)

# NEON 优化
set(CMAKE_C_FLAGS "${CMAKE_C_FLAGS} -mfpu=neon-vfpv4 -mfloat-abi=hard")
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -mfpu=neon-vfpv4 -mfloat-abi=hard")
```

### scripts/deploy.sh

```bash
#!/bin/bash
set -e

# 开发板 IP（根据实际修改）
BOARD_IP="192.168.1.100"
BOARD_USER="root"
DEPLOY_DIR="/opt/endoscope"

SCRIPT_DIR="$(cd "$(dirname "$0")" && pwd)"
PROJECT_DIR="$(dirname "$SCRIPT_DIR")"
BUILD_DIR="$PROJECT_DIR/build_arm"

echo "部署到开发板 $BOARD_IP:$DEPLOY_DIR ..."

# 创建目录
ssh $BOARD_USER@$BOARD_IP "mkdir -p $DEPLOY_DIR/captures $DEPLOY_DIR/videos"

# 拷贝可执行文件
scp "$BUILD_DIR/endoscope_demo" $BOARD_USER@$BOARD_IP:$DEPLOY_DIR/
scp "$BUILD_DIR/test_capture" $BOARD_USER@$BOARD_IP:$DEPLOY_DIR/

# 拷贝标定文件（如果有）
if [ -f "$PROJECT_DIR/camera_calib.yml" ]; then
    scp "$PROJECT_DIR/camera_calib.yml" $BOARD_USER@$BOARD_IP:$DEPLOY_DIR/
fi

# 拷贝运行脚本
scp "$SCRIPT_DIR/run.sh" $BOARD_USER@$BOARD_IP:$DEPLOY_DIR/

# 拷贝 OpenCV 动态库（如果开发板上没有）
# scp /opt/opencv_arm/lib/libopencv_*.so* $BOARD_USER@$BOARD_IP:/usr/lib/

echo "部署完成！"
echo "在开发板上运行: ssh $BOARD_USER@$BOARD_IP '$DEPLOY_DIR/run.sh'"
```

## 7.6 开机自启动（可选）

如果想让开发板上电后自动运行内窥镜程序：

```bash
# 方式一：rc.local
echo "/opt/endoscope/run.sh &" >> /etc/rc.local

# 方式二：systemd service
cat > /etc/systemd/system/endoscope.service << 'EOF'
[Unit]
Description=Endoscope Demo
After=multi-user.target

[Service]
Type=simple
ExecStart=/opt/endoscope/run.sh
Restart=on-failure
RestartSec=5
Environment=QT_QPA_PLATFORM=linuxfb:fb=/dev/fb0

[Install]
WantedBy=multi-user.target
EOF

systemctl enable endoscope
systemctl start endoscope
```
