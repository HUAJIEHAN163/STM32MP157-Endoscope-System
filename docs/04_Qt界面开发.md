# 04 — Qt 界面开发

## 4.1 设计思路

从 VideoMonitor 的 C/S 架构改造为**单机架构**：

```
原 VideoMonitor 架构：
  Server（V4L2采集 → TCP发送） → 网络 → Client（TCP接收 → OpenCV处理 → Qt显示）

新架构：
  V4l2Camera线程（V4L2采集）
      ↓ frameReady 信号
  ImageProcessor（OpenCV处理）
      ↓ 处理后的 QImage
  MainWindow（Qt显示 + 用户交互）
```

### 保留 VideoMonitor 的部分

- QImage 显示机制（QPainter 绘制）
- 图像处理算法的 checkbox 开关机制
- 拍照保存功能
- QImage ↔ cv::Mat 转换工具类（util.h/cpp）

### 去掉的部分

- 整个 TCP 网络层（QTcpServer、QTcpSocket、RecvThread）
- 登录系统（LoginSystemDialog、SQLite）
- 多窗口管理（OneVideo 的多实例机制）
- 服务端程序（VideoMonitorServer 整个目录）

## 4.4 核心代码 — mainwindow.cpp

```cpp
#include "mainwindow.h"
#include "v4l2_camera.h"
#include "util.h"

#include <QPainter>
#include <QVBoxLayout>
#include <QHBoxLayout>
#include <QGroupBox>
#include <QDateTime>
#include <QKeyEvent>
#include <QDir>
#include <QDebug>

MainWindow::MainWindow(QWidget *parent)
    : QMainWindow(parent), m_camera(nullptr),
      m_frozen(false), m_rotateAngle(0),
      m_recording(false), m_frameCount(0),
      m_undistortInited(false)
{
    setupUi();
    setupCamera();

    // FPS 定时器
    m_fpsTimer.start();
    QTimer *fpsTimer = new QTimer(this);
    connect(fpsTimer, &QTimer::timeout, this, &MainWindow::updateFps);
    fpsTimer->start(1000);

    // 创建保存目录
    QDir().mkpath("./captures");
    QDir().mkpath("./videos");
}

MainWindow::~MainWindow() {
    if (m_camera) {
        m_camera->stop();
        m_camera->wait();
    }
    if (m_videoWriter.isOpened())
        m_videoWriter.release();
}

void MainWindow::setupUi() {
    // 全屏或固定大小（适配开发板屏幕）
    setFixedSize(800, 480);
    // setWindowState(Qt::WindowFullScreen);  // 全屏模式

    QWidget *central = new QWidget(this);
    setCentralWidget(central);

    QHBoxLayout *mainLayout = new QHBoxLayout(central);
    mainLayout->setContentsMargins(2, 2, 2, 2);

    // 左侧：视频显示区域（用 paintEvent 绘制，这里只占位）

    videoArea->setMinimumSize(600, 450);
    mainLayout->addWidget(videoArea, 1);

    // 右侧：控制面板
    QVBoxLayout *ctrlLayout = new QVBoxLayout;
    ctrlLayout->setSpacing(5);

    // 图像处理组
    QGroupBox *procGroup = new QGroupBox("图像处理");
    QVBoxLayout *procLayout = new QVBoxLayout(procGroup);
    m_chkClahe = new QCheckBox("CLAHE增强");
    m_chkUndistort = new QCheckBox("畸变校正");
    m_chkDehaze = new QCheckBox("去雾");
    m_chkSharpen = new QCheckBox("锐化");
    m_chkDenoise = new QCheckBox("降噪");
    procLayout->addWidget(m_chkClahe);
    procLayout->addWidget(m_chkUndistort);
    procLayout->addWidget(m_chkDehaze);
    procLayout->addWidget(m_chkSharpen);
    procLayout->addWidget(m_chkDenoise);
    ctrlLayout->addWidget(procGroup);

    // 参数调节组
    QGroupBox *paramGroup = new QGroupBox("参数调节");
    QVBoxLayout *paramLayout = new QVBoxLayout(paramGroup);
    paramLayout->addWidget(new QLabel("亮度"));
    m_sliderBrightness = new QSlider(Qt::Horizontal);
    m_sliderBrightness->setRange(-64, 64);

void MainWindow::setupCamera() {
    m_camera = new V4l2Camera("/dev/video0", 640, 480, 4);
    connect(m_camera, &V4l2Camera::frameReady,
            this, &MainWindow::onFrameReady, Qt::QueuedConnection);
    connect(m_camera, &V4l2Camera::errorOccurred, [this](const QString &msg) {
        m_lblStatus->setText("错误: " + msg);
    });

    if (m_camera->open()) {
        m_camera->start();
        m_lblStatus->setText("状态: 运行中");
    } else {
        m_lblStatus->setText("状态: 摄像头打开失败");
    }
}

void MainWindow::onFrameReady(const QImage &image) {
    if (m_frozen) return;

    m_frameCount++;

    // 图像处理
    QImage processed = processImage(image);

    // 旋转
    if (m_rotateAngle != 0) {
        QTransform transform;
        transform.rotate(m_rotateAngle);
        processed = processed.transformed(transform);
    }

    // 录像
    if (m_recording && m_videoWriter.isOpened()) {
        cv::Mat mat = Util::QImage2cvMat(processed);
        m_videoWriter.write(mat);
    }

    // 缩放到显示区域
    m_displayImage = processed.scaled(m_videoRect.size(),
                                       Qt::KeepAspectRatio,
                                       Qt::SmoothTransformation);
    update();  // 触发 paintEvent
}

void MainWindow::paintEvent(QPaintEvent *event) {
    QMainWindow::paintEvent(event);
    if (m_displayImage.isNull()) return;

    QPainter painter(this);
    // 居中绘制
    int x = m_videoRect.x() + (m_videoRect.width() - m_displayImage.width()) / 2;
    int y = m_videoRect.y() + (m_videoRect.height() - m_displayImage.height()) / 2;
    painter.drawImage(x, y, m_displayImage);

    // 绘制时间戳 OSD
    painter.setPen(Qt::yellow);
    painter.setFont(QFont("Monospace", 10));
    painter.drawText(x + 10, y + 20,
                     QDateTime::currentDateTime().toString("yyyy-MM-dd hh:mm:ss"));
}

void MainWindow::keyPressEvent(QKeyEvent *event) {
    switch (event->key()) {
    case Qt::Key_Escape: close(); break;
    case Qt::Key_Space: onToggleFreeze(); break;
    case Qt::Key_C: onCapturePhoto(); break;
    case Qt::Key_R: onToggleRecord(); break;
    default: QMainWindow::keyPressEvent(event);
    }
}

void MainWindow::onCapturePhoto() {
    if (m_displayImage.isNull()) return;
    QString filename = QString("./captures/%1.jpg")
        .arg(QDateTime::currentDateTime().toString("yyyyMMdd_hhmmss"));
    m_displayImage.save(filename, "JPEG", 95);
    m_lblStatus->setText("拍照: " + filename);
}

void MainWindow::onToggleRecord() {
    if (!m_recording) {
        QString filename = QString("./videos/%1.avi")
            .arg(QDateTime::currentDateTime().toString("yyyyMMdd_hhmmss"));
        m_videoWriter.open(filename.toStdString(),
                           cv::VideoWriter::fourcc('M','J','P','G'),
                           25,
                           cv::Size(m_camera->getWidth(), m_camera->getHeight()));
        if (m_videoWriter.isOpened()) {
            m_recording = true;
            m_btnRecord->setText("⏹ 停止录像");
            m_lblStatus->setText("录像中...");
        }
    } else {
        m_recording = false;
        m_videoWriter.release();
        m_btnRecord->setText("⏺ 录像");
        m_lblStatus->setText("录像已保存");
    }
}

void MainWindow::onToggleFreeze() {
    m_frozen = !m_frozen;
    m_btnFreeze->setText(m_frozen ? "▶ 继续" : "⏸ 冻结");
    m_lblStatus->setText(m_frozen ? "画面已冻结" : "状态: 运行中");
}

void MainWindow::onRotateImage() {
    m_rotateAngle = (m_rotateAngle + 90) % 360;
    m_lblStatus->setText(QString("旋转: %1°").arg(m_rotateAngle));
}

void MainWindow::onBrightnessChanged(int value) {
    if (m_camera) m_camera->setBrightness(value);
}

void MainWindow::onContrastChanged(int value) {
    if (m_camera) m_camera->setContrast(value);
}

void MainWindow::updateFps() {
    double elapsed = m_fpsTimer.elapsed() / 1000.0;
    double fps = m_frameCount / elapsed;
    m_lblFps->setText(QString("FPS: %1").arg(fps, 0, 'f', 1));
    m_frameCount = 0;
    m_fpsTimer.restart();
}
```

## 4.5 processImage 的实现

这里只是调用框架，具体算法实现见下一章（05_内窥镜图像处理算法.md）：

```cpp
QImage MainWindow::processImage(const QImage &input) {
    cv::Mat src = Util::QImage2cvMat(const_cast<QImage&>(input));
    if (src.empty()) return input;

    // 按顺序应用各算法
    if (m_chkClahe->isChecked())
        ImageProcessor::applyCLAHE(src, src);

    if (m_chkUndistort->isChecked() && m_undistortInited)
        cv::remap(src, src, m_map1, m_map2, cv::INTER_LINEAR);

    if (m_chkDehaze->isChecked())
        ImageProcessor::applyDehaze(src, src);

    if (m_chkSharpen->isChecked())
        ImageProcessor::applySharpen(src, src);

    if (m_chkDenoise->isChecked())
        ImageProcessor::applyDenoise(src, src);

    return Util::cvMat2QImage(src);
}
```
